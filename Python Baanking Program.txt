def show_balance(balance):
    print(f"your balance is {balance}")

def deposit():
    amount = float(input("Enter your amount here: "))
    if amount < 0:
        print("This is not a appropreate amount")
        return 0
    else:
        return amount
def withdraw(balance):
    amount = float(input("Enter the withdraw amount here: "))
    if amount > balance:
        print("insuffecent balance")
        return 0
    elif amount < 0:
        print("Your amount is less than Zero")
        return 0
    else:
        return amount
def main():
    balance = 0
    is_running = True

    while is_running:
        
        print("Banking Program")
        print("1.Balance")
        print("2.Deposit")
        print("3.Withdraw")
        print("4.Exit")

        choice = input("Enter any no from 1 to 4: ")

        if choice == '1':
            show_balance(balance)
        elif choice == '2':
            balance += deposit()
        elif choice == '3':
            balance -= withdraw(balance)
        elif choice == '4':
            is_running = False
        else:
            print("This is not valid")

    print("Thank you have a nice day!")
if  __name__ == '__main__':
    main()
